18 class Median():
19     def __call__(self, inputs):
20         stacked = torch.stack(inputs, dim=0)
21         values_upper, _ = stacked.median(dim=0)
22         values_lower, _ = (-stacked).median(dim=0)
23         return (values_upper - values_lower) / 2
